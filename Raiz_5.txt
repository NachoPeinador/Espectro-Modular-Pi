\section*{Título Sugerido para el Programa}

\textbf{«La Música Oculta de $\pi$: Descubriendo los Dos Canales de los Números Primos»}

\section*{Duración}
15-20 minutos

\section*{Puntos Clave para la Entrevista}

\subsection*{1. La Idea Principal (2-3 minutos)}
\begin{itemize}
\item $\pi$ tiene una \textbf{estructura dual} oculta: se descompone en dos ``canales'' de números primos
\item \textbf{Canal 6k+1}: 1, 7, 13, 19, 25... 
\item \textbf{Canal 6k+5}: 5, 11, 17, 23, 29...
\item Casi todos los primos (excepto 2 y 3) viven en estos dos canales
\end{itemize}

\subsection*{2. El Experimento Visual (3-4 minutos)}
\begin{itemize}
\item \textbf{Series de Leibniz transformada:}
\[
\pi = 3\left(1 - \frac{1}{5} + \frac{1}{7} - \frac{1}{11} + \frac{1}{13} - \frac{1}{17} + \cdots\right)
\]
\item \textbf{Demostración visual:} Mostrar cómo desaparecen los múltiplos de 3
\item \textbf{Analogía musical:} Dos instrumentos tocando juntos para crear la melodía de $\pi$
\end{itemize}

\subsection*{3. Conexión con Ramanujan (3-4 minutos)}
\begin{itemize}
\item Las fórmulas rápidas de Ramanujan también siguen este patrón
\item \textbf{Ejemplo:} En la famosa serie, el número 396 = 4×9×11 contiene el primo 11 (que es 6×1+5)
\item \textbf{Curiosidad:} Cada término da 8 dígitos de $\pi$ vs 1 dígito cada millón de términos de Leibniz
\end{itemize}

\subsection*{4. Aplicaciones Sorprendentes (3-4 minutos)}
\begin{itemize}
\item \textbf{Algoritmo BBP:} Calcular cualquier dígito de $\pi$ sin los anteriores
\item \textbf{Supercongruencias:} Patrones ocultos módulo números primos
\item \textbf{Ejemplo concreto:} El primo 17 siempre da residuo 246 módulo 289
\end{itemize}

\subsection*{5. Mensaje Final (2-3 minutos)}
\begin{itemize}
\item $\pi$ no es un número cualquiera: tiene \textbf{estructura interna}
\item Los números primos siguen \textbf{patrones armónicos}
\item Las matemáticas siguen escondiendo \textbf{belleza inesperada}
\item \textbf{Invitación a la audiencia:} ¿Podrán encontrar patrones similares en otras constantes?
\end{itemize}
